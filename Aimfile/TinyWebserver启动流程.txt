# 确认MySQL就绪
mysql -u root -p123

# 启动TinyWebServer
cd ~/TinyWebServer-master
./server


# 确保虚拟机有局域网IP
虚拟机网络必须设置为桥接网络,使主机能直接访问它
ip addr show ens33

若 state DOWN，需手动启用：
sudo ip link set ens33 up
sudo dhclient ens33

# 检查网络连通性
ping 192.168.186.136
