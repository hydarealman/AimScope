获取源码：执行 git clone https://github.com/qinguoyi/TinyWebServer.git 并进入目录

安装依赖：确保系统有 GCC 和 MySQL。在 Ubuntu 上可以执行：
sudo apt update
sudo apt install build-essential mysql-server libmysqlclient-dev



# 创建数据库和表
CREATE DATABASE qgydb;
USE qgydb;
CREATE TABLE user(
    username char(50) NULL,
    passwd char(50) NULL
) ENGINE=InnoDB;
INSERT INTO user(username, passwd) VALUES('test', '123456');
EXIT;
