rosbridge 是一个 ROS 功能包套装（rosbridge_suite），它的作用是把 ROS 的通信（话题/服务/参数）转换成 JSON + WebSocket 协议，让非 ROS 环境（网页浏览器、Python 脚本等）也能直接与 ROS 系统交互。

# 确认ROS版本
echo $ROS_DISTRO

# 安装rosbridge_suite元包
sudo apt-get update
sudo apt-get install ros-noetic-rosbridge-s

# 验证安装
rospack list | grep rosbridge

 # 启动rosbridge WebSocket服务器
 roslaunch rosbridge_server rosbridge_websocket.launch
 
 # 验证 rosbridge 节点
 rosnode list
 
 # Python客户端测试
 # 安装roslibpy
 pip install roslibpy
 
 # 启动HTTP服务器托管页面
 # 进入 panel.html 所在目录
 cd ~/rosbridge_demo
 
 python -m http.server 8000 --bind 0.0.0.0
 
 # 浏览器访问
 http://localhost:8000/panel.html
